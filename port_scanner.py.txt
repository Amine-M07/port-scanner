import socket

def scan_ports(target, start_port, end_port):
    print("-" * 50)
    print(f"Scanning target: {target}")
    print("-" * 50)

    try:
        target_ip = socket.gethostbyname(target)
        print(f"Target IP: {target_ip}\n")
    except socket.gaierror:
        print("Error: Unable to resolve the target hostname.")
        return

    for port in range(start_port, end_port + 1):
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        s.settimeout(1)  # Timeout set to 1 second
        result = s.connect_ex((target_ip, port))
        
        if result == 0:
            print(f"[+] Port {port} is OPEN")
        s.close()

    print("\nScan Complete.")

def main():
    print("Welcome to My Port Scanner!")
    target = input("Enter the target hostname or IP: ")
    start_port = int(input("Enter the starting port number: "))
    end_port = int(input("Enter the ending port number: "))
    
    if start_port > end_port:
        print("Error: Starting port must be less than or equal to the ending port.")
        return
    
    scan_ports(target, start_port, end_port)

if __name__ == "__main__":
    main()
